from bokeh.plotting import figure, show
from bokeh.io import push_notebook, show, output_notebook
from bokeh.charts import Bar, Line
output_notebook()

tooltips=[
    ('read', '@y')
]

box_plot = Bar(df, label=['category'],
               values='download',
               bar_width=0.4,
               agg='count',
               title='data.go.kr 분류별 총합',
               tooltips=tooltips,
               legend=False)

show(box_plot)
